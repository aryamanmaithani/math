\begin{enumerate}[label=(\arabic*)]
	\item For $x, y \in \mathbb{R},$ show that 
	\begin{enumerate*} [label=(\roman*)]
		\item $|xy| = |x| |y|$ \hspace{1 cm}
		\item (Triangle Inqequality) $|x + y| \le |x| + |y|$
	\end{enumerate*}
	\item For $x, y \in \mathbb{R},$ show that $\max\{x, y\} = \dfrac{x + y + |x-y|}{2}.$ Identify a similar relation for $\min\{x, y\}.$
	\item Identify the set $\{x \in \mathbb{R}| |x - 3| < 5\}$ (with proof).
	\item Prove or disprove: For $x, y \in \mathbb{C}, |xy| = |x||y|.$
	\item For a function $f:X\to , A\subset X,$ and $B\subset Y,$ define $f^{-1}(B) = \{a \in X|f(a) \in B\}$ and $f(A) = \{b \in Y|$ there exists $a \in X$ such that $b = f(a)\}.$\\
	Let $f:\mathbb{R}\to\mathbb{R}$ be given by $f(x) = x^2.$
	\begin{enumerate} [nosep]
		\item Find $f(A)$ for $A =$
		\begin{enumerate*}[label=(\roman*)] 
			\item $\left\{1, \frac{1}{2}, \frac{1}{3}, -\frac{1}{2}\right\}$
			\item $[0, 2]$
			\item $(1, 2]$
			\item $[-2, 1)$
		\end{enumerate*}
		\item Find $f^{-1}(B)$ for $B =$
		\begin{enumerate*}[label=(\roman*)] 
			\item $\{4\}$
			\item $\{1\}$
			\item $[0, 1]$
			\item $[-4, 1]$
			\item $(0, 1)$
		\end{enumerate*}
	\end{enumerate}
	What are your answers when 
	\begin{enumerate*}[label=(\roman*)] 
		\item $f(x) = x^3$
		\item $f(x) = \sin(\pi x)?$
	\end{enumerate*}
	\item Let $X$ be the set of $2\times2$ matrices with entries in $\mathbb{R}.$
	\begin{enumerate} [nosep]
		\item Find $f^{-1}(\{\textbf{0}\}),$ where $f:X\to X$ is given by\\
		\begin{enumerate*}[label = (\roman*)] 
			\item $f(M) = M^2$
			\item $f(M) = M^2 - M$
			\item $f(M) = M - M^T$
			\item $f(M) = MM^T.$\\
		\end{enumerate*}
		In (iv), what is $f^{-1}(\{I\})?$
		\item Let $f:X\to\mathbb{R}$ be given by $f(M) = \det(M).$
		\begin{enumerate*}[label = (\roman*)] 
			\item If $A$ is the set of orthogonal matrices, what if $f(A)?$
			\item What if $f^{-1}(\{\textbf{0}\})?$
		\end{enumerate*}
	\end{enumerate}
	\item For $B = \{0\}, \{1\}, [0, 1], (1, 2],$ find $f^{-1}(B),$ where $f:\mathbb{R}^2\to\mathbb{R}$ is given by\\
	$f(x, y) = $
	\begin{enumerate*}[label = (\roman*)] 
		\item $x$
		\item $y$
		\item $x^2 + y^2$
		\item $xy.$
	\end{enumerate*}
	\item Let $f:X\to Y$ be a function, $A, A_1, A_2 \subset X; B, B_1, B_2 \subset Y.$
	\begin{enumerate} [nosep]
		\item If $A \subset A_1,$ show that $f(A) \subset f(A_1).$ Is the same true under inverse images?
		\item Show that $f^{-1}(B_1 \cap B_2) = f^{-1}(B_1) \cap f^{-1}(B_2).$ Is the same true for unions and complements?
		\item For $A_1, A_2 \subset X,$ is one of $f(A_1 \cup A_2)$ and $f(A_1) \cup f(A_2)$ contained in the other? Is the containment proper? When does equality hold? Answer these questions for unions and complements.
		\item What is the relation between $A$ and $f^{-1}(f(A))?$ Is the containment proper? When does equality hold? Answer these questions for $B$ and $f(f^{-1}(B)).$
	\end{enumerate}
	\item Find the limit of the following sequences if they exist, else prove that the sequence diverges.\\
	(i) $a_n = \frac{1}{n^2}$ for all $n\in \mathbb{N}.$ \hfil (ii) $b_n = \frac{1}{n^2}$ for all $n \in \mathbb{N}.$\\	
	(iii) $c_n = n$ for all $n\in \mathbb{N}.$ \hfil (iv) $d_n = (-1)^n$ for all $n \in \mathbb{N}.$
\end{enumerate}

\begin{enumerate}[label = (\arabic*)]
    \item Let $S = \{a \in \mathbb{Z}|a$ is a multiple of $3\}.$\\
    Show that 
    \begin{enumerate*}[label = (\roman*)]
    	\item $0 \in S$
    	\item If $a \in S,$ then $-a \in S.$
    	\item If $a, b \in S,$ then $a + b \in S.$
    \end{enumerate*}
    \item Let $S\subset\mathbb{N}$ be such that
	\begin{enumerate*}[label = (\arabic*)] 
		\item $1 \in S$ and
		\item For $k \in \mathbb{N},$ if $\{1, 2, \cdots, k\} \subset S,$ then $k + 1 \in S.$\\
	\end{enumerate*}
	Show that $S = \mathbb{N}.$
	\item Prove or disprove:\\
	Let $(a_n)$ be a monotonically decreasing sequence of real numbers. Then $(a_n)$ is convergent.
	\item Prove or disprove: The sequence $\left(\dfrac{(-1)^n}{2}\right)$ is convergent.
\end{enumerate}
